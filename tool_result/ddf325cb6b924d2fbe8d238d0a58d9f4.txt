    <script>
        // 存储所有资讯
        let allNews = [];
        let currentPage = 0;
        const newsPerPage = 2;
        const maxPages = 5;

        // 解析内容
        function parseContent() {
            const text = document.getElementById('pasteContent').value;
            if (!text.trim()) {
                alert('请先粘贴内容！');
                return;
            }

            allNews = [];
            const lines = text.split('\n');
            let currentItem = null;

            for (let i = 0; i < lines.length; i++) {
                const line = lines[i].trim();
                
                // 匹配标题行（数字、顿号开头）
                const titleMatch = line.match(/^(\d+)、(.+)$/);
                if (titleMatch) {
                    if (currentItem) {
                        allNews.push(currentItem);
                    }
                    currentItem = {
                        title: titleMatch[2],
                        content: '',
                        source: ''
                    };
                    continue;
                }

                // 匹配来源行
                const sourceMatch = line.match(/^来源：(.+)$/);
                if (sourceMatch && currentItem) {
                    currentItem.source = sourceMatch[1];
                    continue;
                }

                // 其他内容归为正文
                if (currentItem && line && !line.startsWith('来源：')) {
                    if (currentItem.content) {
                        currentItem.content += '\n' + line;
                    } else {
                        currentItem.content = line;
                    }
                }
            }

            // 添加最后一个
            if (currentItem) {
                allNews.push(currentItem);
            }

            if (allNews.length === 0) {
                alert('未解析到有效内容，请检查格式！');
                return;
            }

            // 更新列表显示
            updateNewsList();
            
            // 更新预览
            currentPage = 0;
            updatePreview();
            
            alert(`✅ 成功解析 ${allNews.length} 条资讯，将生成 ${Math.ceil(allNews.length / newsPerPage)} 张图片`);
        }

        // 清空粘贴内容
        function clearPaste() {
            document.getElementById('pasteContent').value = '';
            allNews = [];
            updateNewsList();
            updatePreview();
        }

        // 更新资讯列表
        function updateNewsList() {
            const container = document.getElementById('newsList');
            document.getElementById('newsCount').textContent = allNews.length;

            if (allNews.length === 0) {
                container.innerHTML = '<p style="color:#999;text-align:center;padding:20px;">暂无内容，请粘贴正文后解析</p>';
                return;
            }

            container.innerHTML = allNews.map((news, index) => `
                <div class="news-item-simple">
                    <div class="title">${index + 1}、${news.title}</div>
                    <div class="source">来源：${news.source || '未标注'}</div>
                </div>
            `).join('');
        }

        // 更新预览
        function updatePreview() {
            const title = document.getElementById('mainTitle').value || 'AI 资讯快报';
            const date = document.getElementById('newsDate').value || '2026.04.04';
            const container = document.getElementById('previewContainer');
            const totalPages = Math.max(1, Math.ceil(allNews.length / newsPerPage));

            // 更新页码指示器
            document.getElementById('pageIndicator').textContent = `第 ${currentPage + 1} / ${totalPages} 页`;
            document.getElementById('btnPrev').disabled = currentPage === 0;
            document.getElementById('btnNext').disabled = currentPage >= totalPages - 1 || currentPage >= maxPages - 1;

            // 只预览当前页
            const startIndex = currentPage * newsPerPage;
            const endIndex = Math.min(startIndex + newsPerPage, allNews.length);
            const currentNews = allNews.slice(startIndex, endIndex);

            if (currentNews.length === 0) {
                container.innerHTML = `
                    <div style="text-align:center;color:#999;padding:60px;">
                        <p style="font-size:48px;margin-bottom:20px;">📝</p>
                        <p>暂无内容</p>
                        <p style="font-size:14px;margin-top:10px;">请粘贴正文后点击"解析内容"</p>
                    </div>
                `;
                return;
            }

            container.innerHTML = `
                <div class="preview-card-wrapper">
                    <h3>第 ${currentPage + 1} 页（资讯 ${startIndex + 1}-${endIndex}）</h3>
                    <div id="preview-container">
                        <div id="preview-content">
                            <div class="preview-header">
                                <div class="preview-title">${title}</div>
                                <div class="preview-date">${date}</div>
                            </div>
                            <div class="preview-news-list">
                                ${currentNews.map((news, index) => `
                                    <div class="preview-news-item">
                                        <div class="preview-news-title">${startIndex + index + 1}、${news.title}</div>
                                        <div class="preview-news-content">${news.content}</div>
                                        ${news.source ? `<div class="preview-news-source">来源：${news.source}</div>` : ''}
                                    </div>
                                `).join('')}
                            </div>
                        </div>
                    </div>
                </div>
            `;
        }

        // 上一页
        function prevPage() {
            if (currentPage > 0) {
                currentPage--;
                updatePreview();
            }
        }

        // 下一页
        function nextPage() {
            const totalPages = Math.ceil(allNews.length / newsPerPage);
            if (currentPage < totalPages - 1 && currentPage < maxPages - 1) {
                currentPage++;
                updatePreview();
            }
        }

        // 下载当前页
        function downloadCurrent() {
            if (allNews.length === 0) {
                alert('请先解析内容！');
                return;
            }

            const previewContent = document.getElementById('preview-content');
            const date = document.getElementById('newsDate').value.replace(/\./g, '');
            const pageNum = currentPage + 1;

            html2canvas(previewContent, {
                scale: 2,
                backgroundColor: '#ffffff',
                logging: false,
                useCORS: true
            }).then(canvas => {
                const link = document.createElement('a');
                link.download = `AI 资讯快报_${date}_P${pageNum}.png`;
                link.href = canvas.toDataURL('image/png');
                link.click();
            });
        }

        // 下载全部
        async function downloadAll() {
            if (allNews.length === 0) {
                alert('请先解析内容！');
                return;
            }

            const totalPages = Math.min(Math.ceil(allNews.length / newsPerPage), maxPages);
            const date = document.getElementById('newsDate').value.replace(/\./g, '');

            for (let i = 0; i < totalPages; i++) {
                currentPage = i;
                updatePreview();
                
                // 等待渲染
                await new Promise(resolve => setTimeout(resolve, 300));
                
                const previewContent = document.getElementById('preview-content');
                
                const canvas = await html2canvas(previewContent, {
                    scale: 2,
                    backgroundColor: '#ffffff',
                    logging: false,
                    useCORS: true
                });

                const link = document.createElement('a');
                link.download = `AI 资讯快报_${date}_P${i + 1}.png`;
                link.href = canvas.toDataURL('image/png');
                link.click();
                
                // 等待下载
                await new Promise(resolve => setTimeout(resolve, 500));
            }

            alert(`✅ 已下载 ${totalPages} 张图片！`);
        }

        // 页面加载后初始化
        window.onload = function() {
            updatePreview();
        };
    </script>
