        }

        function updateNewsList() {
            const listEl = document.getElementById('newsList');
            const countEl = document.getElementById('newsCount');
            countEl.textContent = allNews.length;
            if (allNews.length === 0) {
                listEl.innerHTML = '<div style="padding:20px; text-align:center; color:#94a3b8; font-size:13px;">暂无内容</div>'; return;
            }
            listEl.innerHTML = allNews.map((item, i) => `
                <div class="news-list-item">
                    <span class="num">${i + 1}</span>
                    <span class="title">${item.title}</span>
                    <div class="source">来源：${item.source}</div>
                </div>
            `).join('');
        }

        function updatePreview() {
            const contentEl = document.getElementById('preview-content');
            const listContainer = contentEl.querySelector('.preview-news-list');
            const title = document.getElementById('mainTitle').value;
            const date = document.getElementById('newsDate').value;
            
            if (allNews.length === 0) {
                contentEl.innerHTML = `<div style="display:flex; flex-direction:column; align-items:center; justify-content:center; height:100%; color:var(--text-tertiary);"><svg viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="1.5" width="48" height="48"><path d="M14.5 2H6a2 2 0 0 0-2 2v16a2 2 0 0 0 2 2h12a2 2 0 0 0 2-2V7.5L14.5 2z"/><polyline points="14 2 14 8 20 8"/></svg><div style="margin-top:16px; font-size:15px;">暂无预览内容</div></div>`;
                updatePageControls(1); return;
            }

            const totalPages = Math.ceil(allNews.length / itemsPerPage);
            const start = currentPage * itemsPerPage;
            const pageNews = allNews.slice(start, start + itemsPerPage);
            
            // 单条模式添加 class 用于居中
            const singleModeClass = itemsPerPage === 1 ? 'single-mode' : '';

            let newsHtml = pageNews.map((item, i) => `
                <div class="preview-news-item">
                    <div class="preview-news-number">${start + i + 1}</div>
                    <div class="preview-news-title">${item.title}</div>
                    <div class="preview-news-content">${item.content}</div>
                    <div class="preview-news-source">来源：${item.source}</div>
                </div>
            `).join('');
            
            contentEl.innerHTML = `
                <div class="preview-header">
                    <div class="preview-title">${title}</div>
                    <div class="preview-meta">
                        <span class="preview-date">${date}</span>
                        <span class="preview-source-tag">AI 每日精选</span>
                    </div>
                </div>
                <div class="preview-news-list ${singleModeClass}">${newsHtml}</div>
            `;
            updatePageControls(totalPages);
        }

        function updatePageControls(totalPages) {
            document.getElementById('pageIndicator').textContent = `第 ${currentPage + 1} / ${totalPages} 页`;
            document.getElementById('prevBtn').disabled = currentPage === 0;
            document.getElementById('nextBtn').disabled = currentPage >= totalPages - 1;
        }

        function prevPage() { if (currentPage > 0) { currentPage--; updatePreview(); } }
        function nextPage() { const totalPages = Math.ceil(allNews.length / itemsPerPage); if (currentPage < totalPages - 1) { currentPage++; updatePreview(); } }
        function clearPaste() { document.getElementById('pasteContent').value = ''; allNews = []; currentPage = 0; updateNewsList(); updatePreview(); }
        function clearAll() { clearPaste(); document.getElementById('mainTitle').value = 'AI 资讯快报'; document.getElementById('newsDate').value = new Date().toISOString().split('T')[0].replace(/-/g, '.'); }
        function manualParse() { const content = document.getElementById('pasteContent').value.trim(); if (content) parseContent(content); }

        async function downloadCurrent() {
            if (allNews.length === 0) return;
            const previewContent = document.getElementById('preview-content');
            const date = document.getElementById('newsDate').value.replace(/\./g, '');
            const pageNum = currentPage + 1;
            html2canvas(previewContent, { scale: 2, backgroundColor: null, logging: false, useCORS: true }).then(canvas => {
                const link = document.createElement('a');
                link.download = `AI_资讯_${date}_P${pageNum}.jpg`;
                link.href = canvas.toDataURL('image/jpeg', 0.95);
                link.click();
            });
        }

        async function downloadAll() {
            if (allNews.length === 0) return;
            const totalPages = Math.ceil(allNews.length / itemsPerPage);
            const date = document.getElementById('newsDate').value.replace(/\./g, '');
            const originalPage = currentPage;
            for (let i = 0; i < totalPages; i++) {
                currentPage = i; updatePreview();
                await new Promise(resolve => setTimeout(resolve, 300));
                const previewContent = document.getElementById('preview-content');
                const canvas = await html2canvas(previewContent, { scale: 2, backgroundColor: null, logging: false, useCORS: true });
                const link = document.createElement('a');
                link.download = `AI_资讯_${date}_P${i + 1}.jpg`;
                link.href = canvas.toDataURL('image/jpeg', 0.95);
                link.click();
                await new Promise(resolve => setTimeout(resolve, 500));
            }
            currentPage = originalPage; updatePreview();
            alert(`已下载 ${totalPages} 张图片`);
        }

        window.onload = function() {
            toggleTheme(); // 初始化随机风格
            updatePreview();
        };
